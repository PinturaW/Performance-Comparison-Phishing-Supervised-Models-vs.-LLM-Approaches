<script>
    const cardNumberInput = document.getElementById('cardNumberInput');
    const expiryInput = document.getElementById('expiryInput');
    const cvvInput = document.getElementById('cvvInput');

    const dynamicDigits = document.getElementById('dynamicDigits');
    const cardNumber = document.getElementById('cardNumber');
    const dynamicExpiry = document.getElementById('dynamicExpiry');
    const expiryDate = document.getElementById('expiryDate');
    const dynamicCVV = document.getElementById('dynamicCVV');
    const cvvCode = document.getElementById('cvvCode');

    function updateDigits() {
        let value = cardNumberInput.value.replace(/\D/g, '').slice(0, 10);
        let formatted = '';

        for (let i = 0; i < 10; i++) {
            formatted += value[i] ? value[i] : '_';
            if (i === 1 || i === 5) formatted += ' ';
        }
        dynamicDigits.textContent = formatted;

        if (value.length === 10) {
            cardNumber.style.color = '#28a745';
            cardNumber.style.textShadow = '0 0 12px rgba(40,167,69,1)';
            if (!expiryInput.value) expiryInput.focus();
        } else {
            cardNumber.style.color = 'white';
            cardNumber.style.textShadow = '0 0 8px rgba(255,255,255,0.7)';
        }
    }

    function updateExpiry() {
        let value = expiryInput.value.replace(/\D/g, '').slice(0, 4);
        let formatted = '';

        if (value.length >= 2) {
            formatted = value.slice(0, 2) + '/' + value.slice(2);
        } else {
            formatted = value;
        }

        while (formatted.length < 5) formatted += '_';

        dynamicExpiry.textContent = formatted;

        if (value.length === 4) {
            expiryDate.style.color = '#28a745';
            expiryDate.style.textShadow = '0 0 10px rgba(40,167,69,1)';
            if (!cvvInput.value) cvvInput.focus();
        } else {
            expiryDate.style.color = 'white';
            expiryDate.style.textShadow = '0 0 6px rgba(255,255,255,0.6)';
        }
    }

    function updateCVV() {
        let value = cvvInput.value.replace(/\D/g, '').slice(0, 3);
        let formatted = '';

        for (let i = 0; i < 3; i++) {
            formatted += value[i] ? value[i] : '_';
        }

        dynamicCVV.textContent = formatted;

        if (value.length === 3) {
            cvvCode.style.color = '#28a745';
            cvvCode.style.textShadow = '0 0 10px rgba(40,167,69,1)';
        } else {
            cvvCode.style.color = 'white';
            cvvCode.style.textShadow = '0 0 6px rgba(255,255,255,0.6)';
        }
    }

    window.onload = () => cardNumberInput.focus();
</script>
