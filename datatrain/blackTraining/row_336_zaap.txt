            function initializeSwiper(sliderType) {
                $("." + sliderType + "-slider").each(function () {
                    try {
                        
                    
                    const $this = $(this);
                    const sliderId = $this.data('id');
                    if (!sliderId) return;
                    
                    const containerSelector = '.' + sliderType + '-slider-' + sliderId;
                    
                    const containerEl = document.querySelector(containerSelector);
                    console.log("sliderType",sliderType,"containerEl",containerEl)
                    if (!containerEl) return;

                    const leftArrow = document.querySelector("." + sliderType + '-left-arrow-' + sliderId);
                    const rightArrow = document.querySelector("." + sliderType + '-right-arrow-' + sliderId);
                    const leftFade = document.querySelector("." + sliderType + '-left-fade-' + sliderId);
                    const rightFade = document.querySelector("." + sliderType + '-right-fade-' + sliderId);
                    const swiper = new Swiper(containerSelector, {
                        slidesPerView: 'auto',
                        spaceBetween: sliderType == "instagram" ? 0 : 15,
                        freeMode: true,
                        navigation: {
                            nextEl: rightArrow,
                            prevEl: leftArrow,
                        },
                        on: {
                            init: function (swiperInstance) {
                                updateFadeVisibility(swiperInstance);
                            },
                            slideChange: function (swiperInstance) {
                                updateFadeVisibility(swiperInstance);
                            },
                            reachEnd: function (swiperInstance) {
                                updateFadeVisibility(swiperInstance);
                            },
                            reachBeginning: function (swiperInstance) {
                                updateFadeVisibility(swiperInstance);
                            },
                            fromEdge: function (swiperInstance) {
                                updateFadeVisibility(swiperInstance);
                            }
                        }
                    });

                    function updateFadeVisibility(swiperInstance) {
                        const isBeginning = swiperInstance.isBeginning;
                        const isEnd = swiperInstance.isEnd;

                        if (leftArrow) leftArrow.style.opacity = isBeginning ? '0' : '1';
                        if (leftFade) leftFade.style.display = isBeginning ? 'none' : 'block';

                        if (rightArrow) rightArrow.style.opacity = isEnd ? '0' : '1';
                        if (rightFade) rightFade.style.display = isEnd ? 'none' : 'block';
                    }} catch (error) {
                        console.log("error in initializeSwiper",sliderType,error)
                    }
                });
            }
            //  Initialize all sliders
            $(document).ready(function () {

                    "folder-portfolio",
                    "youtube",
                    "instagram",
                    "spotify",
                    "instagram"
                ];

                sliderTypes.forEach(type => initializeSwiper(type));
            });
