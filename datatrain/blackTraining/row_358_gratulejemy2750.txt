        async function callGemini(prompt, buttonElement, responseElementId) {
            const buttonSpan = buttonElement.querySelector('span');
            const buttonSpinner = buttonElement.querySelector('svg');
            const originalButtonText = buttonSpan.textContent;
            const responseElement = document.getElementById(responseElementId);

            buttonSpinner.classList.remove('hidden');
            buttonSpan.textContent = 'Oluşturuluyor...';
            buttonElement.disabled = true;
            responseElement.innerHTML = '<div class="flex justify-center items-center p-6"><div class="loader"></div></div>'; // Show loader

            try {
                let chatHistory = [{ role: "user", parts: [{ text: prompt }] }];
                const payload = { contents: chatHistory };
                const apiKey = ""; // Provided by the environment
                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;
                
                const apiResponse = await fetch(apiUrl, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });

                if (!apiResponse.ok) throw new Error(`API isteği başarısız: ${apiResponse.status}`);
                
                const result = await apiResponse.json();

                if (result.candidates && result.candidates[0]?.content?.parts[0]) {
                    let text = result.candidates[0].content.parts[0].text;
                    // Basic Markdown to HTML
                    text = text.replace(/(\r\n|\n|\r)/gm, "<br>");
                    text = text.replace(/\*\*(.*?)\*\*/g, '<h3>$1</h3>');
                    text = text.replace(/\* (.*?)(<br>|$)/g, '<li>$1</li>');
                    text = text.replace(/(<li>.*<\/li>)/g, '<ul>$1</ul>').replace(/<\/ul><br><ul>/g, '');
                    responseElement.innerHTML = `<div class="p-4 bg-gray-50 rounded-md border text-left">${text}</div>`;
                } else {
                    throw new Error("API'den geçerli bir yanıt alınamadı.");
                }
            } catch (error) {
                console.error("Gemini API Hatası:", error);
                responseElement.innerHTML = `<p class="p-4 bg-red-100 text-red-700 rounded-md">Bir hata oluştu: ${error.message}</p>`;
            } finally {
                buttonSpinner.classList.add('hidden');
                buttonSpan.textContent = originalButtonText;
                buttonElement.disabled = false;
            }
        }

        function getFlagEmoji(countryCode) {
            if (!countryCode) return '';
            const codePoints = countryCode
                .toUpperCase()
                .split('')
                .map(char => 0x1F1E6 + (char.charCodeAt(0) - 'A'.charCodeAt(0)));
            return String.fromCodePoint(...codePoints);
        }
